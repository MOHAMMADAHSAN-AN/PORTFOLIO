Face Recognition Attendance System

A real-time attendance management system powered by InsightFace deep learning (ArcFace) for face detection and recognition. Built on Django REST Framework with SQLite — students are automatically marked present when their face is detected through a live camera feed with anti-spoofing protection.

Completed

// Key Features

  • Real-time face detection with InsightFace (ArcFace) at 640×640 resolution using CPUExecutionProvider
  • Anti-spoofing: FFT screen pattern detection, Laplacian texture analysis, reflection & glare detection
  • Face embedding stored as binary (pickle) in SQLite — cosine similarity matching with 0.4 threshold
  • Django REST Framework with CSRF-exempt custom session authentication
  • IST timezone-aware attendance logging via pytz (Asia/Kolkata)
  • Django admin panel with custom list_display for User and Attendance models
  • Liveness detection: landmark motion score + face center stability checks

# attendance/views.py — Face Recognition Attendance System from django.shortcuts import render from django.views.decorators.csrf import csrf_exempt from django.utils.decorators import method_decorator from rest_framework.views import APIView from rest_framework.response import Response from rest_framework import status from rest_framework.authentication import SessionAuthentication, BasicAuthentication from .models import User, Attendance import numpy as np import pickle, cv2 from insightface.app import FaceAnalysis # Initialize InsightFace with CPU provider app = FaceAnalysis(providers=['CPUExecutionProvider']) app.prepare(ctx_id=0, det_size=(640, 640)) def embedding_to_binary(embedding): return pickle.dumps(embedding) def binary_to_embedding(binary): return pickle.loads(binary) def load_image_to_rgb_filelike(file_obj): file_obj.seek(0) nparr = np.frombuffer(file_obj.read(), np.uint8) img = cv2.imdecode(nparr, cv2.IMREAD_COLOR) if img is None: raise ValueError("Failed to decode image") return cv2.cvtColor(img, cv2.COLOR_BGR2RGB) def detect_anti_spoofing(img): """Single image anti-spoofing: screen patterns + texture + glare""" faces = app.get(img) if not faces: return False, "No face detected" best_face = max(faces, key=lambda f: (f.bbox[2]-f.bbox[0])*(f.bbox[3]-f.bbox[1]) if f.bbox is not None else 0) bbox = best_face.bbox # 1. Laplacian texture sharpness check x1,y1,x2,y2 = [int(v) for v in bbox] face_region = cv2.cvtColor(img[y1:y2,x1:x2], cv2.COLOR_RGB2GRAY) laplacian_var = float(cv2.Laplacian(face_region, cv2.CV_64F).var()) if laplacian_var < 50.0: return False, f"Low texture (lap={laplacian_var:.1f})" # 2. FFT screen pattern detection f = np.fft.fft2(face_region); fshift = np.fft.fftshift(f) mag = np.abs(fshift) mid_energy = np.sum(mag) / mag.size if mid_energy > 1.8: return False, "Screen pattern detected" return True, "Anti-spoofing passed" # ─── Custom auth (bypass CSRF) ─── class CsrfExemptSessionAuthentication(SessionAuthentication): def enforce_csrf(self, request): return @method_decorator(csrf_exempt, name='dispatch') class UserRegisterView(APIView): authentication_classes = (CsrfExemptSessionAuthentication, BasicAuthentication) def post(self, request): user_id = request.data.get('user_id') name = request.data.get('name') face_image = request.FILES.get('face_image') if not all([user_id, name, face_image]): return Response({'error':'All fields required'}, status=status.HTTP_400_BAD_REQUEST) img = load_image_to_rgb_filelike(face_image) is_real, reason = detect_anti_spoofing(img) if not is_real: return Response({'error':'Anti-spoofing failed','detail':reason}, status=400) faces = app.get(img) embedding = faces[0].embedding user, created = User.objects.get_or_create( user_id=user_id, defaults={'name':name,'embedding':embedding_to_binary(embedding)} ) if not created: user.name=name; user.embedding=embedding_to_binary(embedding); user.save() return Response({'message':'User registered successfully'}, status=201) @method_decorator(csrf_exempt, name='dispatch') class PunchAttendanceView(APIView): authentication_classes = (CsrfExemptSessionAuthentication, BasicAuthentication) def post(self, request): face_image = request.FILES.get('face_image') if not face_image: return Response({'error':'Face image required'}, status=400) img = load_image_to_rgb_filelike(face_image) is_real, reason = detect_anti_spoofing(img) if not is_real: return Response({'error':'Spoofing detected','detail':reason}, status=400) new_embedding = app.get(img)[0].embedding best_match, highest_sim = None, 0.0 for user in User.objects.all(): stored = binary_to_embedding(user.embedding) sim = float(np.dot(stored, new_embedding) / (np.linalg.norm(stored) * np.linalg.norm(new_embedding))) if sim > highest_sim and sim > 0.4: highest_sim = sim; best_match = user if best_match: Attendance.objects.create(user=best_match) return Response({ 'message': f'Attendance recorded for {best_match.name}', 'confidence': f'{highest_sim:.2%}', 'user_id': best_match.user_id, 'name': best_match.name }, status=200) return Response({'error':'User not recognized'}, status=404) def attendance_page(request): return render(request, 'attendance.html')

Heart Disease Prediction System

A clinical decision-support web app that predicts heart disease likelihood from 13 patient health parameters. Trained using Logistic Regression on the UCI Cleveland Heart Disease dataset — deployed with a Flask web interface and JSON prediction logging for audit trails.

Completed

// Key Features

  • Logistic Regression trained on 13 UCI clinical features (age, cholesterol, thalach, ECG, etc.)
  • Stratified 80/20 train-test split with random_state=2 for reproducibility
  • Model serialized to heart_model.pkl (pickle) for Flask deployment
  • Flask POST /predict endpoint with full form-based HTML UI
  • Every prediction auto-logged to predictions.json with timestamp + all feature values
  • Full Jupyter Notebook included: EDA → training → evaluation workflow
  • Result: "Heart Disease" or "No Heart Disease" displayed in-browser

# app.py — Heart Disease Prediction Flask App from flask import Flask, request, render_template import numpy as np import pickle, json, os from datetime import datetime app = Flask(__name__) model = pickle.load(open("heart_model.pkl", "rb")) PREDICTION_FILE = "predictions.json" if not os.path.exists(PREDICTION_FILE): with open(PREDICTION_FILE, "w") as f: json.dump([], f) @app.route("/") def home(): return render_template("index.html") @app.route("/predict", methods=["POST"]) def predict(): data = request.form features = { "age": float(data["age"]), # Patient age "sex": float(data["sex"]), # 1=Male, 0=Female "cp": float(data["cp"]), # Chest pain type (0-3) "trestbps": float(data["trestbps"]), # Resting blood pressure "chol": float(data["chol"]), # Serum cholesterol mg/dl "fbs": float(data["fbs"]), # Fasting blood sugar >120 "restecg": float(data["restecg"]), # Resting ECG results "thalach": float(data["thalach"]), # Max heart rate achieved "exang": float(data["exang"]), # Exercise induced angina "oldpeak": float(data["oldpeak"]), # ST depression "slope": float(data["slope"]), # Slope of peak ST segment "ca": float(data["ca"]), # Major vessels colored "thal": float(data["thal"]), # Thalassemia type } input_array = np.array(list(features.values())).reshape(1, -1) prediction = model.predict(input_array)[0] result = "No Heart Disease" if prediction == 0 else "Heart Disease" # Audit log: save every prediction to JSON log_entry = { "timestamp": datetime.now().isoformat(), "input_features": features, "prediction": int(prediction), "result": result } with open(PREDICTION_FILE, "r+") as f: data = json.load(f); data.append(log_entry) f.seek(0); json.dump(data, f, indent=4) return render_template("index.html", prediction_text=result) if __name__ == "__main__": app.run(debug=True)
# model.py — Heart Disease ML Training Pipeline import numpy as np import pandas as pd import pickle from sklearn.model_selection import train_test_split from sklearn.linear_model import LogisticRegression # Load UCI Cleveland Heart Disease Dataset (303 rows, 14 columns) heart_data = pd.read_csv("heart_disease_data.csv") X = heart_data.drop(columns='target', axis=1) # 13 clinical features Y = heart_data['target'] # 0 = healthy, 1 = disease # Stratified split preserves class distribution in both sets X_train, X_test, Y_train, Y_test = train_test_split( X, Y, test_size=0.2, stratify=Y, random_state=2 ) # Train: 242 samples | Test: 61 samples model = LogisticRegression(max_iter=1000) # max_iter=1000 ensures convergence model.fit(X_train, Y_train) # Evaluate train_acc = model.score(X_train, Y_train) test_acc = model.score(X_test, Y_test) print(f"Train accuracy: {train_acc:.3f} | Test accuracy: {test_acc:.3f}") # Serialize model for Flask deployment pickle.dump(model, open("heart_model.pkl", "wb")) print("✅ Model saved as heart_model.pkl")
